// Initialize with your API key
const db = new Digibase('/api/data').setToken('pk_your_key_here');

// Fetch authors with related books
const authors = await db.from('authors')
                        .include('books')
                        .get();

// Each author has a nested 'books' array!
authors.forEach(author => {
    console.log(`${author.name} wrote ${author.books?.length || 0} books`);
});
